    attribute name : String +private;
  }
  
  [@150,215] 
  class Permission {
    attribute name : String +private;
  }

  [@vertical]
  association permissionsOfRole {
    role role : Role +shared;
    role permissions : Permission +navigable;
  }
  
  [@vertical]
  association usersOfGroup {
    role group : Group +shared;
    role users : User +navigable;
  }

  [@vertical]
  association rolesOfUser {
    role user : User;
    role roles : Role +navigable;
  }
